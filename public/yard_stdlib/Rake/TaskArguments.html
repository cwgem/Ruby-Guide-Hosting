def initialize(names, values, parent=nil)
  @names = names
  @parent = parent
  @hash = {}
  names.each_with_index { |name, i|
    @hash[name.to_sym] = values[i] unless values[i].nil?
  }
end

def lookup(name)
  if @hash.has_key?(name)
    @hash[name]
  elsif @parent
    @parent.lookup(name)
  end
end

def new_scope(names)
  values = names.collect { |n| self[n] }
  self.class.new(names, values, self)
end
