def format_datetime(time)
  if @datetime_format.nil?
    time.strftime("%Y-%m-%dT%H:%M:%S.") << "%06d " % time.usec
  else
    time.strftime(@datetime_format)
  end
end

def msg2str(msg)
  case msg
  when ::String
    msg
  when ::Exception
    "#{ msg.message } (#{ msg.class })\n" <<
      (msg.backtrace || []).join("\n")
  else
    msg.inspect
  end
end
