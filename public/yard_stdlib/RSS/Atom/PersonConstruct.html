def self.append_features(klass)

  klass.class_eval do

     ["name", nil],
     ["uri", "?"],
     ["email", "?"],
    ].each do |tag, occurs|
      install_have_attribute_element(tag, URI, occurs, nil, :content)
    end
  end
end
