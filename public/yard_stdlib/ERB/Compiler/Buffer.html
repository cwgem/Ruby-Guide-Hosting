def initialize(compiler, enc=nil)
  @compiler = compiler
  @line = []
  @script = enc ? "#coding:#{enc.to_s}\n" : ""
  @compiler.pre_cmd.each do |x|
    push(x)
  end
end

def close
  return unless @line
  @compiler.post_cmd.each do |x|
    push(x)
  end
  @script << (@line.join('; '))
  @line = nil
end
