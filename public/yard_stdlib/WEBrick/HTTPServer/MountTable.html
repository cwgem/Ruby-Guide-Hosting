def compile
  k = @tab.keys
  k.sort!
  k.reverse!
  k.collect!{|path| Regexp.escape(path) }
  @scanner = Regexp.new("^(" + k.join("|") +")(?=/|$)")
end

def normalize(dir)

  ret
end
