def parse(str)
  parser = NQXML::StreamingParser.new(str)
  parser.each do |ele|
    case ele
    when NQXML::Text
      @data = ele.text
      #character(ele.text)
    when NQXML::Tag
      if ele.isTagEnd
        endElement(ele.name)
      else
        startElement(ele.name, ele.attrs)
      end
    end
  end # do
end
