def nOE_element_def(element, append = nil)
  s = <<-END      attributes={attributes=>nil} if attributes.kind_of?(String)
      "<#{element.upcase}
" + attributes.collect{|name, value|
        next unless value
        " " + CGI::escapeHTML(name.to_s) +
        if true == value
          ""
        else
          '="' + CGI::escapeHTML(value.to_s) + '"'
        end
      }.join + ">"
  END
  s.sub!(/\Z/, " +") << append if append
  s
end
