def self.filter_backtrace bt # :nodoc:
  return ["No backtrace"] unless bt

  new_bt = []

  unless $DEBUG then
    bt.each do |line|
      break if line.rindex MINI_DIR, 0
      new_bt << line
    end

    new_bt = bt.reject { |line| line.rindex MINI_DIR, 0 } if new_bt.empty?
    new_bt = bt.dup if new_bt.empty?
  else
    new_bt = bt.dup
  end

  new_bt
end
