def call(*args)
  if @pid == $$
    path, tmpfile = *@data

    STDERR.print "removing ", path, "..." if $DEBUG

    tmpfile.close if tmpfile

    # keep this order for thread safeness
    if path
      File.unlink(path) if File.exist?(path)
    end

    STDERR.print "done\n" if $DEBUG
  end
end
