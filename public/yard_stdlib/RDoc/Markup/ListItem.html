def accept visitor
  visitor.accept_list_item_start self

  @parts.each do |part|
    part.accept visitor
  end

  visitor.accept_list_item_end self
end

def pretty_print q # :nodoc:
  q.group 2, '[item: ', ']' do
    if @label then
      q.text @label
      q.breakable
    end

    q.seplist @parts do |part|
      q.pp part
    end
  end
end
