def dup
  sibling = self.class.new
  instance_variables.each do |ivar|
    value = self.instance_variable_get(ivar)
    new_value = value.clone rescue value
    sibling.instance_variable_set(ivar, new_value)
  end
  sibling.taint if tainted?
  sibling
end
