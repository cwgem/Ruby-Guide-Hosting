def self.append_features(klass)
  super

  ::RSS::DublinCoreModel::ELEMENT_NAME_INFOS.each do |name, plural_name|
    plural_name ||= "#{name}s"
    full_name = "#{RSS::DC_PREFIX}_#{name}"
    full_plural_name = "#{RSS::DC_PREFIX}_#{plural_name}"
    plural_klass_name = "DublinCore#{Utils.to_class_name(plural_name)}"
    klass.def_classed_elements(full_name, "value", plural_klass_name,
                               full_plural_name, name)
    klass.module_eval(<<-EOC      def new_#{full_name}(value=nil)
        _#{full_name} = #{full_plural_name}.new_#{name}
        _#{full_name}.value = value
        if block_given?
          yield _#{full_name}
        else
          _#{full_name}, __FILE__, __LINE__ + 1)

        end
      end
    EOC
  end

  klass.module_eval(<<-EOC    # For backward compatibility
    alias #{DC_PREFIX}_rightses #{DC_PREFIX}, __FILE__, __LINE__ + 1)
_rights_list
  EOC
end

def self.install_dublin_core(klass)
  ::RSS::DublinCoreModel::ELEMENT_NAME_INFOS.each do |name, plural_name|
    plural_name ||= "#{name}s"
    klass_name = Utils.to_class_name(name)
    full_klass_name = "DublinCore#{klass_name}"
    plural_klass_name = "DublinCore#{Utils.to_class_name(plural_name)}"
    klass.module_eval(<<-EOC    class #{plural_klass_name} < #{plural_klass_name}Base
      class #{full_klass_name} < #{full_klass_name}Base
      end
      #{klass_name} = #{full_klass_name}, __FILE__, __LINE__ + 1)

    end
EOC
  end
end
