Overview

An abstract class for enumerating pseudo-prime numbers.

Concrete subclasses should override succ, next, rewind.

def each(&block)
  return self.dup unless block
  if @ubound
    last_value = nil
    loop do
      prime = succ
      break last_value if prime > @ubound
      last_value = block.call(prime)
    end
  else
    loop do
      block.call(succ)
    end
  end
end

def with_object(obj)

  each do |prime|
    yield prime, obj
  end
